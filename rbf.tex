%Round Bottome Flask
\documentclass[preview]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc}

\begin{document}
	\begin{tikzpicture}
		\pgfmathsetmacro{\R}{1}		%radius of flask
		\pgfmathsetmacro{\neck}{3}	%number of necks
		\pgfmathsetmacro{\pos}{90}	%position of first neck in degrees
		\pgfmathsetmacro{\ang}{45}	%separation of necks in degrees

		%clockwise from 90
		\coordinate (a) at ({\pos-atan((24/200)/\R)}:\R+40/100) [fill=none]{};
		\coordinate (b) at ({\pos-atan((24/200)/\R)}:\R) [fill=none]{};
		\coordinate (c) at ({45+atan((24/200)/\R)}:\R) {};
		\coordinate (d) at ({45+atan((24/200)/\R)}:\R+40/100) {};
		\coordinate (e) at ({45-atan((24/200)/\R)}:\R+40/100) {};
		\coordinate (f) at ({45-atan((24/200)/\R)}:\R) {};
		\coordinate (g) at ({135+atan((24/200)/\R)}:\R) {};
		\coordinate (h) at ({135+atan((24/200)/\R)}:\R+40/100) {};
		\coordinate (i) at ({135-atan((24/200)/\R)}:\R+40/100) {};
		\coordinate (j) at ({135-atan((24/200)/\R)}:\R) {};
		\coordinate (k) at ({\pos+atan((24/200)/\R)}:\R) {};
		\coordinate (l) at ({\pos+atan((24/200)/\R)}:\R+40/100) {};

		\draw (a) -- (b) arc ({90-atan((24/200)/\R)}:{45+atan((24/200)/\R)}:\R) -- (d);
		\draw (e) -- (f) arc ({45-atan((24/200)/\R)}:{135+atan((24/200)/\R)-360}:\R) -- (h);
		\draw (i) -- (j) arc ({135-atan((24/200)/\R)}:{90+atan((24/200)/\R)}:\R) -- (l);
	\end{tikzpicture}
\end{document}
